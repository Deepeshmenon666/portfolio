    <script>
        // EVENT 1: Typewriter Effect
        const nameText = "Deepesh";
        const bioText = "> IT Student & Aspiring Developer_";
        let nameIdx = 0;
        let bioIdx = 0;

        function typeEffect() {
            if (nameIdx < nameText.length) {
                document.querySelector('#typewriter-name span').textContent += nameText.charAt(nameIdx);
                nameIdx++;
                setTimeout(typeEffect, 150);
            } else if (bioIdx < bioText.length) {
                document.getElementById('typewriter-bio').textContent += bioText.charAt(bioIdx);
                bioIdx++;
                setTimeout(typeEffect, 50);
            }
        }
        window.onload = typeEffect;

        // EVENT 2: Theme Toggle
        const themeBtn = document.getElementById('theme-toggle');
        themeBtn.addEventListener('click', () => {
            document.body.classList.toggle('light-mode');
            themeBtn.innerText = document.body.classList.contains('light-mode') ? "☀️" : "🌙";
        });

        // EVENT 3: Scroll Navbar
        window.addEventListener('scroll', () => {
            const nav = document.getElementById('main-nav');
            window.scrollY > 50 ? nav.classList.add('nav-scrolled') : nav.classList.remove('nav-scrolled');
        });

        // EVENT 4: About Text Glow
        const aboutText = document.getElementById('about-text');
        aboutText.addEventListener('mouseover', () => aboutText.classList.add('active-glow'));
        aboutText.addEventListener('mouseleave', () => aboutText.classList.remove('active-glow'));

        // EVENT 5: Skill & Project Hover Effects
        const cards = document.querySelectorAll('.skill-card, .project-card');
        cards.forEach(card => {
            card.addEventListener('mouseenter', () => {
                card.style.borderColor = 'var(--accent)';
                card.style.boxShadow = '0 10px 25px var(--accent-glow)';
                card.style.transform = card.classList.contains('skill-card') ? 'translateY(-5px)' : 'translateY(-10px)';
            });
            card.addEventListener('mouseleave', () => {
                card.style.borderColor = '';
                card.style.boxShadow = 'none';
                card.style.transform = 'translateY(0)';
            });
        });

        // EVENT 6: Form Submit
        const form = document.getElementById('contact-form');
        form.addEventListener('submit', (e) => {
            e.preventDefault();
            const email = document.getElementById('user-email').value;
            const msg = document.getElementById('response-msg');
            msg.style.color = 'var(--accent)';
            msg.innerText = "Success! Deepesh will contact you at " + email;
            form.reset();
        });

        // EVENT 7: KEYBOARD SHORTCUTS
        window.addEventListener('keydown', (e) => {
            if (e.ctrlKey) {
                const key = e.key.toLowerCase();
                let targetId = "";

                if (key === 'a') targetId = "about";
                if (key === 's') targetId = "skill";
                if (key === 'p') targetId = "projects";
                if (key === 'c') targetId = "contact";

                if (targetId) {
                    e.preventDefault(); 
                    const targetElement = document.getElementById(targetId);
                    if (targetElement) {
                        targetElement.scrollIntoView({ behavior: 'smooth' });
                    }
                }
            }
        });
    </script>
